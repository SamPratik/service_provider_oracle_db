DECLARE
CURSOR CUSTOMER_WORKER_cursor
IS
SELECT Customer_id, Customer_name,Service_Date,WORKER_NID,NAME FROM CUSTOMER NATURAL JOIN Takes_Service NATURAL JOIN WORKER;
CUSTOMER_WORKER CUSTOMER_WORKER_cursor%ROWTYPE;
BEGIN
OPEN CUSTOMER_WORKER_cursor ;
FETCH CUSTOMER_WORKER_cursor INTO CUSTOMER_WORKER;
CLOSE CUSTOMER_WORKER_cursor ;
END;
